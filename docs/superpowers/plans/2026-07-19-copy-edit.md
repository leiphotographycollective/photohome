### Task 3: No-em-dash guard test for content files

**Files:**
- Create: `tests/no-em-dash.test.ts`

**Interfaces:**
- Consumes: all exports of `@/content/pricing`, `@/content/homepage`, `@/content/experience`, `@/content/second-weddings`, `@/content/site`.
- Produces: a permanent guard; later tasks must not introduce em dashes into content-file strings.

- [ ] **Step 1: Write the test**

Create `tests/no-em-dash.test.ts` with exactly:

```ts
import { describe, expect, it } from "vitest";
import * as pricing from "@/content/pricing";
import * as homepage from "@/content/homepage";
import * as experience from "@/content/experience";
import * as secondWeddings from "@/content/second-weddings";
import * as site from "@/content/site";

// Photo alt text ("a") and asset paths are out of scope for the no-em-dash
// rule (see the 2026-07-19 copy-edit spec); everything else is rendered copy.
const SKIP_KEYS = new Set(["a", "path"]);

function collectStrings(
  value: unknown,
  out: string[],
  seen = new Set<object>()
): void {
  if (typeof value === "string") {
    out.push(value);
    return;
  }
  if (typeof value !== "object" || value === null) return;
  if (seen.has(value)) return;
  seen.add(value);
  if (Array.isArray(value)) {
    for (const v of value) collectStrings(v, out, seen);
    return;
  }
  for (const [k, v] of Object.entries(value)) {
    if (SKIP_KEYS.has(k)) continue;
    collectStrings(v, out, seen);
  }
}

describe("content files contain no em dashes in rendered copy", () => {
  const modules: Record<string, object> = {
    pricing,
    homepage,
    experience,
    "second-weddings": secondWeddings,
    site,
  };
  for (const [name, mod] of Object.entries(modules)) {
    it(`${name} strings are em-dash free`, () => {
      const strings: string[] = [];
      collectStrings({ ...mod }, strings);
      expect(strings.length).toBeGreaterThan(0); // walker sanity check
      for (const s of strings) {
        expect(s, `em dash found in ${name}: "${s}"`).not.toContain("—");
      }
    });
  }
});
```

- [ ] **Step 2: Run it**

Run: `npx vitest run tests/no-em-dash.test.ts`
Expected: PASS (5 tests). The content files are already clean after Task 2; the `toBeGreaterThan(0)` assertion proves the walker is actually visiting strings, so a silent no-op cannot masquerade as a pass. If any test FAILS, the message names the offending string; fix that content string (colon/semicolon/comma per Global Constraints) rather than weakening the test.
